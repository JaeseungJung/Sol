pragma solidity 0.8.6;

      contract Test {
          string name = "vitalik";
          event SetName(string name, address sender);
          
          function setName(string memory _name) public {
              name = _name;
              emit SetName(_name, msg.sender);
          }
              
          function getName() public view returns (string memory) {
              return name;
          }
      }
    

      async function init() {

        if (window.ethereum) {
            web3 = new Web3(window.ethereum);
            try {
              // Request account access if needed
              await window.ethereum.enable();
              // Acccounts now exposed
              web3.eth.getAccounts().then(function(accounts){
                account = accounts[0]
              })
            } catch (error) {
            }
          }
          // Legacy dapp browsers...
          else if (window.web3) {
            // Use Mist/MetaMask's provider.
            web3 = window.web3;
            console.log("Injected web3 detected.");
          }
        }
    

      let abi = [
      {
        "anonymous": false,
        "inputs": [
          {
            "indexed": false,
            "internalType": "string",
            "name": "name",
            "type": "string"
          },
          {
            "indexed": false,
            "internalType": "address",
            "name": "sender",
            "type": "address"
          }
        ],
        "name": "SetName",
        "type": "event"
      },
      {
        "inputs": [],
        "name": "getName",
        "outputs": [
          {
            "internalType": "string",
            "name": "",
            "type": "string"
          }
        ],
        "stateMutability": "view",
        "type": "function"
      },
      {
        "inputs": [
          {
            "internalType": "string",
            "name": "_name",
            "type": "string"
          }
        ],
        "name": "setName",
        "outputs": [],
        "stateMutability": "nonpayable",
        "type": "function"
      }
    ]
      contract = new web3.eth.Contract(abi, "0xAB14DA43978aE2287A88e4BF541dCb791D0F9391")

      // listen event
      contract.events.SetName({}, function(error, event){ 
        console.log(event)
      })
    

      function getName(){
        contract.methods.getName().call()
        .then(function(result){
          document.getElementById("name").innerText = result
        });
      }
    

      function setName() {
        contract.methods.setName("Sooyoung").send(
          {from: account}
        )
        .then(function(receipt){
            // receipt can also be a new contract instance, when coming from a "contract.deploy({...}).send()"
            getName()
        });
    
      }
